\begin{UseCase}{CUN4}{Descargar archivo cifrado}{Descargar un archivo cifrado del servicio de almacenamiento (Nube) junto con la llave secreta del usuario que este tiene almacenada en la nube}

	\UCitem{Versión}{1.0 - 15/04/17}
	\UCitem{Autor}{Eder Jonathan Aguirre Cruz}
	\UCitem{Prioridad}{Media}
	\UCitem{Módulo}{Servidor de Llaves}
	\UCitem{Actor}{Usuario}
	\UCitem{Propósito}{Entregar al usuario archivos que desea obtener para un uso posterior. }
	\UCitem{Entradas}{ Nombre archivo a descargar}
	\UCitem{Salidas}{ \begin{itemize}
				\item  Archivo cifrado \textit{C1}
				\item  Archivo cifrado \textit{C2}
			 \end{itemize} }
	\UCitem{Precondiciones}{
					\begin{itemize}			
						\item El usuario debe tener el nombre del archivo que desea descargar
						\item El archivo a descargar debe estar almacenado en la nube						
					\end{itemize}}
	\UCitem{Postcondiciones}{El archivo estará descargado para que el usuario pueda descifrarlo}
	\UCitem{Reglas del negocio}{
		%\begin{itemize}	
		%	\item \BRref{RN-E1}{Alumno inscrito} 
		%	\item \BRref{RN-E4}{Calificación} 
		%	\item \BRref{RN-E6}{Calificación registrada}
		%\end{itemize}
		}
	\UCitem{Mensajes}{
		\begin{itemize}	
			\item \MSGref{MSG1}{Operación exitosa}
			\item  \MSGref{MSG - N1}{Archivo no encontrado}	
		\end{itemize}
		}
\end{UseCase}


% ------------ Trayectoria principal, poner referencias a pantallas del sistema, reglas entre otras cosas que sean necesarias. ------------
\begin{UCtrayectoria}{Principal}
	\UCpaso [\UCactor] Selecciona de su lista de archivos en la nube el nombre del archivo que desea descargar.
	\UCpaso Recibe el nombre del archivo a descargar. 
	\UCpaso Realiza la busqueda del archivo asociado al nombre que recibió. \Trayref{A} 
	\UCpaso Envía los archivos encontrados \textit{C1} y \textit{C2}.
	\UCpaso [\UCactor] Recibe los archivos \textit{C1} y \textit{C2} asociados al nombre que selecciono.
	\UCpaso Muestra el mensaje \MSGref{MSG1}{Operación exitosa}.	
\end{UCtrayectoria}

% ---------------- Trayectorias alternativas -------------- Colocar los mensajes  {\bf MSG1-}

		
\begin{UCtrayectoriaA}{A}{Archivo inexistente}
	\UCpaso Detecta la inexistencia del nombre recibido por el usuario. 
	\UCpaso  Muestra el mensaje \MSGref{MSG - N1}{Archivo no encontrado}.	
	\UCpaso[\UCactor]  Da clic en el botón \IUbuttonAceptar
	\UCpaso Termina el caso de uso.
\end{UCtrayectoriaA}
