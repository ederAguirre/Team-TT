\begin{UseCase}{CUE1}{Consultar calificaciones semestrales actuales}{El alumno podrá visualizar las calificaciones ordinarias y extraordinarias de todas las materias que haya inscrito en el semestre actual, por cada materia podrá además ver el grupo al cual pertenece, el profesor que le impartió esa clase.}

	\UCitem{Versión}{2.0 - 27/11/16}
	\UCitem{Autor}{Jhonatan Saulés Cortés}
	\UCitem{Prioridad}{Media}
	\UCitem{Módulo}{Evaluación}
	\UCitem{Actor}{Alumno}
	\UCitem{Propósito}{Que el alumno visualice sus calificaciones y corrobore que las calificaciones ingresadas al sistema correspondan con las que su profesor le dio a conocer dentro del aula de clases}
	\UCitem{Entradas}{}
	\UCitem{Salidas}{Tabla con las materias inscritas en el semestre actual con su respectivo grupo, nombre de la materia, profesor que imparte la clase, la calificación de cada materia y tipo de calificación}
	\UCitem{Precondiciones}{El alumno debe tener materias inscritas en el semestre actual. El ciclo escolar no debe haber concluido}
	\UCitem{Postcondiciones}{}
	\UCitem{Reglas del negocio}{
		\begin{itemize}	
			\item \BRref{RN-E1}{Alumno inscrito} 
			\item \BRref{RN-E4}{Calificación} 
			\item \BRref{RN-E6}{Calificación registrada}
		\end{itemize}
		}
	\UCitem{Mensajes}{
		\begin{itemize}	
			\item \MSGref{MSG-E8}{Calificaciones no registradas}
			\item \MSGref{MSG-E9}{Materias no inscritas}
		\end{itemize}
		}
\end{UseCase}


% ------------ Trayectoria principal, poner referencias a pantallas del sistema, reglas entre otras cosas que sean necesarias. ------------
\begin{UCtrayectoria}{Principal}
	\UCpaso[\UCactor] Selecciona la opción Calificaciones del semestre.
	\UCpaso Verifica que el alumno haya inscrito materias en el semestre actual \Trayref{A} \Trayref{B} .
	\UCpaso Despliegue la la pantalla \IUref{IUE7}{Calificaciones del semestre}.
	\UCpaso[\UCactor] Consulta las calificaciones del semestre actual.
\end{UCtrayectoria}
